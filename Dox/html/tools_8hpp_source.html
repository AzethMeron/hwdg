34 template<typename ... Args>
35 std::string string_format(const std::string& format, Args ... args)
36 {
37 int size_s = std::snprintf(nullptr, 0, format.c_str(), args ...) + 1; // Extra space for '\0'
38 if (size_s <= 0) { throw std::runtime_error("Error during formatting."); }
39 auto size = static_cast<size_t>(size_s);
40 std::unique_ptr<char[]> buf(new char[size]);
41 std::snprintf(buf.get(), size, format.c_str(), args ...);
42 return std::string(buf.get(), buf.get() + size - 1); // We don't want the '\0' inside
